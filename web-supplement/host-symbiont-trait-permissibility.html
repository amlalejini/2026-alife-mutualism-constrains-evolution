Supplemental Material for Mutualism can constrain host diversity and evolution

Chapter 10 Host-symbiont trait permissibility

Repeat differential function rewards experiment, except vary whether hosts and symbionts are restricted to expressing a single type of function or can express any number of different functions. This controls whether an organism is required to specialize on a function type or is allowed to be a generalist.

10.2 Load and preprocess data

tasks <- c("NOT", "NAND", "OR_NOT", "AND", "OR", "AND_NOT", "NOR", "XOR", "EQU")
max_pop_size <- 100 * 100
threshold <- 0.025 * max_pop_size

Load summary data.

summary_data_path <- paste(
  exp_directory,
  "data",
  "summary.csv",
  sep = "/"
)
summary_data <- read_csv(summary_data_path)
# Do any universal preprocessing
summary_data <- summary_data %>%
  mutate(
    SEED = as.factor(SEED),
    task_credit = paste0(
      "H", HOST_ONLY_FIRST_TASK_CREDIT,
      "S", SYM_ONLY_FIRST_TASK_CREDIT
    ),
    sym_condition = case_when(
      START_MOI == "0" ~ "no-syms",
      START_MOI == "1" & HEALTH_TYPE == "parasite" ~ "parasite",
      START_MOI == "1" & HEALTH_TYPE == "mutualist" ~ "mutualist",
      .default = "unknown" # precaution against unexpected parameterizations
    )
  ) %>%
  mutate(
    task_credit = as.factor(task_credit),
    sym_condition = factor(
      sym_condition,
      levels = c("no-syms", "parasite", "mutualist")
    )
  )

Load time series data.

time_data_path <- paste(
  exp_directory,
  "data",
  "time_series.csv",
  sep = "/"
)
time_data <- read_csv(time_data_path)

# Do any universal preprocessing
time_data <- time_data %>%
  mutate(
    SEED = as.factor(SEED),
    sym_condition = case_when(
      START_MOI == "0" ~ "no-syms",
      START_MOI == "1" & HEALTH_TYPE == "parasite" ~ "parasite",

      .default = "unknown" # precaution against unexpected parameterizations
    ),
    task_credit = paste0(
      "H", HOST_ONLY_FIRST_TASK_CREDIT,
      "S", SYM_ONLY_FIRST_TASK_CREDIT
    )
  ) %>%
  mutate(
    sym_condition = factor(
      sym_condition,
      levels = c("no-syms", "parasite", "mutualist")
    ),
    task_credit = as.factor(task_credit)
  )

10.3 Extinctions

Did host or symbiont populations go extinct in any runs?

sym_min_pop_sizes <- time_data %>%
  group_by(SEED) %>%
  slice_min(
    OrgCounts_hosted_sym_count,
    n = 1
  ) %>%
  slice_min(
    update,
    n = 1
  )

reps_with_extinct_syms <- sym_min_pop_sizes %>%
  group_by(
    sym_condition,
    task_credit
  ) %>%
  summarize(
    reps_with_extinct_syms = sum(OrgCounts_hosted_sym_count == 0),
    total_reps = n()
  )

host_min_pop_sizes <- time_data %>%
  group_by(SEED) %>%
  slice_min(
    OrgCounts_host_count,
    n = 1
  ) %>%
  slice_min(
    update,
    n = 1
  )

reps_with_extinct_hosts <- host_min_pop_sizes %>%
  group_by(
    sym_condition,
    task_credit
  ) %>%
  summarize(
    reps_with_extinct_hosts = sum(OrgCounts_host_count == 0),
    total_reps = n()
  )

extinctions <- full_join(
  reps_with_extinct_syms,
  reps_with_extinct_hosts
)
extinctions <- extinctions %>%
  arrange(
    desc(reps_with_extinct_hosts), desc(reps_with_extinct_syms)
  )
extinction_table <- kable(extinctions) %>%
  kable_styling(latex_options = "striped")
save_kable(
  extinction_table,
  paste(
    plot_dir,
    "extinctions.pdf",
    sep = "/"
  )
)

10.4 Host function performance

First, quantify which functions have been performed over the course of an entire run.

# Pivot task data long, selecting only relevant columns
host_over_time_parent_task_data <- time_data %>%
  select(
    SEED,
    sym_condition,
    task_credit,
    update,
    CurUpdate_AND_NOT_in_host_parent_org_counts,
    CurUpdate_AND_in_host_parent_org_counts,
    CurUpdate_EQU_in_host_parent_org_counts,
    CurUpdate_NAND_in_host_parent_org_counts,
    CurUpdate_NOR_in_host_parent_org_counts,
    CurUpdate_NOT_in_host_parent_org_counts,
    CurUpdate_OR_NOT_in_host_parent_org_counts,
    CurUpdate_OR_in_host_parent_org_counts,
    CurUpdate_XOR_in_host_parent_org_counts
  ) %>%
  pivot_longer(
    c(
      CurUpdate_AND_NOT_in_host_parent_org_counts,
      CurUpdate_AND_in_host_parent_org_counts,
      CurUpdate_EQU_in_host_parent_org_counts,
      CurUpdate_NAND_in_host_parent_org_counts,
      CurUpdate_NOR_in_host_parent_org_counts,
      CurUpdate_NOT_in_host_parent_org_counts,
      CurUpdate_OR_NOT_in_host_parent_org_counts,
      CurUpdate_OR_in_host_parent_org_counts,
      CurUpdate_XOR_in_host_parent_org_counts
    ),
    names_to = "task",
    values_to = "task_in_parent_count"
  ) %>%
  mutate(
    task = str_replace(task, "CurUpdate_", "")
  ) %>%
  mutate(
    task = str_replace(task, "_in_host_parent_org_counts", "")
  ) %>%
  mutate(
    task_completed_by_parent = task_in_parent_count >= threshold,
    task = factor(
      task,
      levels = tasks,
      labels = tasks
    )
  ) %>%
  mutate(
    task_complexity = case_when(
      task == "NOT" ~ 1,
      task == "NAND" ~ 1,
      task == "OR_NOT" ~ 2,
      task == "AND" ~ 2,
      task == "OR" ~ 3,
      task == "AND_NOT" ~ 3,
      task == "NOR" ~ 4,
      task == "XOR" ~ 4,
      task == "EQU" ~ 5,
      .default = -1
    )
  )

host_parent_tasks_all_time_by_task <- host_over_time_parent_task_data %>%
  group_by(
    SEED,
    sym_condition,
    task_credit,
    task
  ) %>%
  summarize(
    n = n(),
    parent_task_completed_ever = any(task_completed_by_parent)
  ) %>%
  ungroup() %>%
  group_by(
    sym_condition,
    task_credit,
    task
  ) %>%
  summarize(
    n = n(),
    parent_tasks_completed_ever = sum(parent_task_completed_ever == TRUE)
  )

Next, summarize function complexity level.

host_max_task_complexity_data <- host_over_time_parent_task_data %>%
  group_by(
    SEED,
    sym_condition,
    task_credit,
    task,
    task_complexity
  ) %>%
  summarize(
    n = n(),
    parent_task_completed_ever = any(task_completed_by_parent)
  ) %>%
  ungroup() %>%
  # This filter works fine because all replicates start with a task
  filter(
    parent_task_completed_ever
  ) %>%
  group_by(
    SEED,
    sym_condition,
    task_credit
  ) %>%
  summarize(
    max_task_complexity = max(task_complexity)
  ) %>%
  mutate(
    sym_condition = factor(
      sym_condition,
      levels = c(
        "no-syms",
        "parasite",
        "mutualist"
      )
    )
  )

10.4.1 Host function performance - all time

host_parent_tasks_all_time_by_task_plt <- host_parent_tasks_all_time_by_task %>%
  ggplot(
    aes(
      x = task,
      color = sym_condition,
      fill = sym_condition,
      y = parent_tasks_completed_ever

    legend.position = "bottom",
